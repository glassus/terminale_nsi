        def codage_rle(liste_octets):
            '''Renvoie une liste d'octets obtenue par compression RLE'''
            liste_rle = []
            i = 0
            while i < len(liste_octets):
                c = liste_octets[i]
                k = 1
                while i+k < len(liste_octets) and liste_octets[i+k] == c:
                    k += 1
                q = k // 255
                r = k % 255
                for _ in range(q):

                    liste_rle.append(c)
                if r != 0:

                i += k
